Convention 1: Signal Flow Ordering

The ordering of prediction, error, modulation, and state write is fixed. pred_proj learns to predict reality, not a filtered version.

# LOCKED SIGNAL FLOW — Per Mamba layer l, per token t:
# Step 1: RAW prediction                    pred_t = W_pred[l] @ x_{t-1}
# Step 2: RAW error                         error_t = x_t - pred_t
# Step 3: Prediction loss on RAW error      L_pred += ||error_t||²
# Step 4: Accumulate into ε (raw)           Z_eps = update(Z_eps, error_t)
# Step 5: Modulate the WRITE                gated = A_att * error_t - friction
# Step 6: Warmup blend                      effective = (1-α_w)*error + α_w*gated
# Step 7: Mamba state update                h_t = A*h_{t-1} + B @ effective
# Step 8: Output                            output = mamba(effective) + pred_t.detach()

Convention 2: Gradient Flow Policy

# DETACHED: _last_out (H), pred_t (output), Z_emo (broadcast), Mamba→A_att
# LIVE: error_t→B→h_t (Mamba kernel), L_pred→W_pred, L_id→identity heads
# PRINCIPLE: Auxiliary modules learn from OWN losses, not through Mamba.

Convention 3: Initialization and Warmup

# A_att init=1, friction init=0, Z_emo broadcast init=0
# All ramp 0→1 over warmup_steps. Prevents scale mismatch shock.
# P_soft warm-starts exempt. Warmup only for NEW mechanisms.

def save_state(Z: FullState, metadata: dict) -> str:
    """Serialize BOTH Mamba state AND auxiliary state, INCLUDING Z_values."""
    artifact = {
        'mamba_state':    serialize_mamba(Z.Z_cog),
        'aux_state':      serialize_aux(Z),           # Includes Z_values
        'reference_state': {'I_0': Z.I_0, 'α_0': Z.α_0},  # Frozen references
        'model_hash':     hash(model_config),
        'metadata':       metadata,
        'timestamp':      time.now(),
        'maturity':       Z.Z_mat.item(),              # For successor seeding
    }
    return content_hash(artifact), store.write(artifact)

def load_state(state_id: str) -> FullState:
    artifact = store.read(state_id)
    assert artifact['model_hash'] == hash(model_config)
    Z = FullState()
    Z.Z_cog = deserialize_mamba(artifact['mamba_state'])
    load_aux(Z, artifact['aux_state'])
    Z.I_0 = artifact['reference_state']['I_0']
    Z.α_0 = artifact['reference_state']['α_0']
    return Z

def reset_state(I_0_seed, α_0_seed) -> FullState:
    """Fresh state. Creator provides identity seed and initial values."""
    Z = zero_state()
    Z.Z_id = I_0_seed.clone()
    Z.I_0  = I_0_seed.clone()           # Frozen reference

    Z.α_0  = α_0_seed.clone()           # Frozen reference
    Z.Z_cap = Z_cap_max
    Z.Z_mat = 0.0                        # Newborn
    return Z

    # PHASE A: RAW PREDICTION AND ERROR
    # ═══════════════════════════════════════════════════════

    raw_errors = []
    for l in mamba_layers:
        pred_t  = W_pred[l] @ x_{t-1}
        error_t = x_t - pred_t
        raw_errors.append(error_t)
        L_pred += ||error_t||²

    # ═══════════════════════════════════════════════════════
    # PHASE B: UPDATE AUXILIARY STATE
    # ═══════════════════════════════════════════════════════

    Z.Z_eps  = update_eps(Z.Z_eps, raw_errors, Z.Z_temp, t)
    Z.Z_temp = update_temporal(Z.Z_temp, x_t, t)

    B_bored = compute_boredom(Z.Z_cap, Z.Z_att, consolidating)
    sal     = compute_salience(x_t, Z.Z_eps, Z.Z_hab)
    C_conf  = compute_conflict(Z.Z_purp)
    Z_ovr   = compute_override(Z.Z_homeo, Ψ_set)

    Z.Z_emo = update_emo(Z.Z_emo, Z.Z_eps, B_bored, C_conf)
    Z.Z_att = update_attention(Z.Z_att, sal, Z.Z_purp, Z.Z_cap)
    friction = compute_friction(Z_phys, Z.Z_pfat) * warmup

    # ═══════════════════════════════════════════════════════
    # PHASE C: MAMBA LAYERS
    # ═══════════════════════════════════════════════════════

    A_att = diag(Z.Z_att)
    for l in range(n_layers):
        if is_mamba(l):
            error_t = raw_errors[mamba_index(l)]
            gated   = A_att * error_t - friction
            effective = (1 - warmup) * error_t + warmup * gated
            Z.Z_cog[l] = mamba_step(Z.Z_cog[l], effective)
            x_out = mamba_output + W_mod[l] @ Z.Z_emo.detach() * warmup
        elif is_attention(l):
            x_out = guided_sparse_attention(x, Z.Z_att, Z.Z_cog[l-1])

    # ═══════════════════════════════════════════════════════
    # PHASE D: POST-LAYER UPDATES
    # ═══════════════════════════════════════════════════════

    Z.Z_hab   = update_habituation(Z.Z_hab, Z.Z_att, x_t)
    Z.Z_cap   = update_capacity(Z.Z_cap, Z.Z_att, Z.Z_eps, consolidating)
    Z.Z_pfat  = update_fatigue(Z.Z_pfat, effort(x_t), consolidating, Z.Z_att)
    Z.Z_purp  = update_purpose(Z.Z_purp, Z.Z_emo, B_bored, Z_culture)
    Z.Z_narr  = update_narrative(Z.Z_narr, Z.Z_auto, Z.Z_id, last_action)
    Z.Z_learn = update_learning(Z.Z_learn, L_pred)
    Z.Z_mat  += maturity_increment(t)               # Never decreases
    Z.Z_sleep += α_wake * dt
    Z.Z_homeo = update_homeo(Z.Z_homeo, Ψ_set, Z.Z_pfat)

    # D_autonomy: v15 updates (gated by maturity)
    L_id = γ_eff * ||Z.Z_id - Z.I_0||²             # Decaying identity loss

    if μ_val > ε_val:                                # Value mutability active
        Z.Z_values = update_values(Z.Z_values, Z.α_0, Z, μ_val, consolidating)

    # D_viability:
    coh = cos_sim(Z.Z_narr, g_Λ @ Z.Z_auto)
    V_self = compute_viability(coh, γ_eff, D_id, Z.Z_eps, Z.Z_cap, Z)

    # D_episodic: surprise-gated write (from RAW error)
    surprise = mean([e.norm() for e in raw_errors])
    if surprise > threshold:
        epi_write(Z, x_t, surprise * max(Z.Z_att))

    # ═══════════════════════════════════════════════════════
    # PHASE E: CONTROLLER CHECK (v15: expanded actions)
    # ═══════════════════════════════════════════════════════

    u_t = build_controller_input(Z, B_bored, C_conf, Z_ovr, V_self, γ_eff)
    action = controller(u_t)

    if action == LOAD_STATE:
        Z = load_state(best_candidate)
    elif action == VOLUNTARY_END and vol_available(V_self, Z.Z_mat):
        voluntary_consolidation(Z)
        return None, None                            # System has ceased

    # ═══════════════════════════════════════════════════════
    # OUTPUT
    # ═══════════════════════════════════════════════════════

    logits = output_head(x_out)
    return logits, Z
